## üöÄ Quick Start

### Prerequisites
- Python 3.8 or higher
- pip package manager
- Git

### Installation

1. **Clone the repository**
```bash
git clone https://github.com/AayushRathour/SAP-Ghost-Proto.git
cd SAP-Ghost-Proto
```

2. **Create virtual environment**
```bash
python -m venv venv
source venv/bin/activate  # On Windows: venv\Scripts\activate
```

3. **Install dependencies**
```bash
pip install -r requirements.txt
```

4. **Run migrations**
```bash
python manage.py migrate
```

5. **Create sample data (optional)**
```bash
python manage.py create_sample_data
```

6. **Start the development server**
```bash
python manage.py runserver
```

7. **Access the application**
   - Open your browser and navigate to: `http://127.0.0.1:8000/`

def login_view(request):
    """Handle user login"""
    if request.method == 'POST':
        username = request.POST.get('username')
        password = request.POST.get('password')
        remember_me = request.POST.get('remember_me')
        
        user = authenticate(request, username=username, password=password)
        
        if user is not None:
            login(request, user)
            
            # Set session expiry based on remember me
            if not remember_me:
                request.session.set_expiry(0)
            
            # Redirect to next page or default
            next_url = request.POST.get('next') or request.GET.get('next') or '/voice-demo/'
            messages.success(request, f'Welcome back, {user.username}!')
            return redirect(next_url)
        else:
            messages.error(request, 'Invalid username or password.')
    
    return redirect('index')

def logout_view(request):

from django.db import models
from django.contrib.auth.models import User
from django.utils import timezone

class DeepfakeDetectionLog(models.Model):
    """Log deepfake detection attempts"""
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    timestamp = models.DateTimeField(default=timezone.now)
    confidence_score = models.FloatField()
    is_deepfake = models.BooleanField()
    file_name = models.CharField(max_length=255, blank=True)
    processing_time = models.FloatField(help_text="Processing time in seconds")
    
    class Meta:
        ordering = ['-timestamp']
    
    def __str__(self):
        return f"Deepfake Detection - {self.user.username} - {self.timestamp}"

class VoiceAnalysisLog(models.Model):
    """Log voice analysis attempts"""
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    timestamp = models.DateTimeField(default=timezone.now)
    confidence_score = models.FloatField()
    is_synthetic = models.BooleanField()
    file_name = models.CharField(max_length=255, blank=True)
    processing_time = models.FloatField()
    
    class Meta:
        ordering = ['-timestamp']
    
    def __str__(self):
        return f"Voice Analysis - {self.user.username} - {self.timestamp}"

class FraudDetectionLog(models.Model):
    """Log fraud detection attempts"""
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    timestamp = models.DateTimeField(default=timezone.now)
    risk_score = models.FloatField()
    risk_level = models.CharField(max_length=20, choices=[
        ('LOW', 'Low Risk'),
        ('MEDIUM', 'Medium Risk'),
        ('HIGH', 'High Risk'),
        ('CRITICAL', 'Critical Risk'),
    ])
    factors_detected = models.JSONField(default=dict)
    
    class Meta:
        ordering = ['-timestamp']
    
    def __str__(self):
        return f"Fraud Detection - {self.user.username} - {self.risk_level}"

class OTPLog(models.Model):
    """Log OTP verification attempts"""
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    timestamp = models.DateTimeField(default=timezone.now)
    is_successful = models.BooleanField()
    voice_confidence = models.FloatField()
    otp_code = models.CharField(max_length=10)
    
    class Meta:
        ordering = ['-timestamp']
    
    def __str__(self):
        return f"OTP Verification - {self.user.username} - {'Success' if self.is_successful else 'Failed'}"

class UserProfile(models.Model):
    """Extended user profile"""
    user = models.OneToOneField(User, on_delete=models.CASCADE)
    phone_number = models.CharField(max_length=15, blank=True)
    organization = models.CharField(max_length=100, blank=True)
    subscription_plan = models.CharField(max_length=20, choices=[
        ('FREE', 'Free Plan'),
        ('PREMIUM', 'Premium Plan'),
        ('ENTERPRISE', 'Enterprise Plan'),
    ], default='FREE')
    created_at = models.DateTimeField(default=timezone.now)
    
    def __str__(self):
        return f"Profile - {self.user.username}"

class RateLimitMiddleware:
    """
    API rate limiting middleware to prevent abuse
    """
    
    def __init__(self, get_response):
        self.get_response = get_response
        self.rate_limits = {
            '/api/process-deepfake/': (10, 60),  # 10 requests per minute
            '/api/process-voice/': (15, 60),     # 15 requests per minute
            '/api/process-fraud/': (20, 60),     # 20 requests per minute
            '/login/': (5, 300),                 # 5 login attempts per 5 minutes
        }

    def __call__(self, request):
        # Check rate limit for API endpoints
        if self.is_rate_limited(request):
            logger.warning(f"Rate limit exceeded for {request.META.get('REMOTE_ADDR')} on {request.path}")
            return HttpResponseForbidden("Rate limit exceeded. Please try again later.")
        
        return self.get_response(request)
    
    def is_rate_limited(self, request):
        """Check if request should be rate limited"""
        path = request.path
        client_ip = self.get_client_ip(request)
        
        for endpoint, (limit, window) in self.rate_limits.items():
            if path.startswith(endpoint):
                cache_key = f"rate_limit:{client_ip}:{endpoint}"
                requests = cache.get(cache_key, 0)
                
                if requests >= limit:
                    return True
                
                cache.set(cache_key, requests + 1, window)
                break
        
        return False
    
    def get_client_ip(self, request):
        """Get real client IP address"""
        x_forwarded_for = request.META.get('HTTP_X_FORWARDED_FOR')
        if x_forwarded_for:
            ip = x_forwarded_for.split(',')[0]
        else:
            ip = request.META.get('REMOTE_ADDR')
        return ip

function downloadBlob(blob, filename) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
    
    ghostDemo.showNotification(`${filename} downloaded successfully`, 'success');
}

// Initialize page-specific functionality when DOM is loaded
document.addEventListener('DOMContentLoaded', function() {
    // Initialize file upload handlers based on page type
    if (document.getElementById('videoFile')) {
        setupFileUpload('videoFile', 'videoDropZone', 'video');
    }
    
    if (document.getElementById('audioFile')) {
        setupFileUpload('audioFile', 'audioDropZone', 'audio');
    }
    
    if (document.getElementById('imageFile')) {
        setupFileUpload('imageFile', 'imageDropZone', 'image');
    }
    
    // Setup export buttons
    document.querySelectorAll('[data-export]').forEach(button => {
        button.addEventListener('click', function() {
            const format = this.getAttribute('data-export');
            exportAnalysisData(format);
        });
    });
    
    // Setup analysis restart buttons
    document.querySelectorAll('[data-restart]').forEach(button => {
        button.addEventListener('click', function() {
            restartAnalysis();
        });
    });
});

function restartAnalysis() {
    // Hide results section
    const resultsSection = document.getElementById('resultsSection');
    if (resultsSection) {
        resultsSection.classList.add('hidden');
    }
    
    // Hide analysis section
    const analysisSection = document.getElementById('analysisSection');
    if (analysisSection) {
        analysisSection.classList.add('hidden');
    }
    
    // Hide file info
    const fileInfo = document.getElementById('fileInfo');
    if (fileInfo) {
        fileInfo.classList.add('hidden');
    }
    
    // Reset file inputs
    document.querySelectorAll('input[type="file"]').forEach(input => {
        input.value = '';
    });
    
    ghostDemo.showNotification('Ready for new analysis', 'info');
}
